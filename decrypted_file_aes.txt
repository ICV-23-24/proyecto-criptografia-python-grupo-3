CREATE TABLE PROVEEDOR (
    codProv NUMBER(4) PRIMARY KEY,
    NombreProv VARCHAR2(25) NOT NULL,
    gerente VARCHAR2(20) NOT NULL
)TABLESPACE TiendaCrecimientoLento STORAGE (INITIAL 1K NEXT 4K);

CREATE TABLE PRODUCTO (
    codProducto NUMBER(4) PRIMARY KEY,
    NombreComercial VARCHAR2(30) NOT NULL,
    nombreInt VARCHAR2(10) NOT NULL,
    Tipo CHAR(1) CHECK(Tipo IN ('C', 'S', 'P', 'D', 'F')),
    Iva NUMBER(4) NOT NULL,
    prCoste FLOAT NOT NULL,
    prVenta FLOAT NOT NULL
)TABLESPACE TiendaCrecimientoConstante STORAGE (INITIAL 3K NEXT 10K);

CREATE TABLE EMPLEADO (
    dniEmpl VARCHAR2(9) PRIMARY KEY,
    Nombre VARCHAR2(12) NOT NULL,
    prApellido VARCHAR2(20) NOT NULL,
    sgApellido VARCHAR2(20),
    Cuenta NUMBER(20) NOT NULL,
    Direccion VARCHAR2(15) NOT NULL,
    Telefono NUMBER(9) NOT NULL,
    Correo VARCHAR2(20) NOT NULL,
    Ciudad VARCHAR2(10) NOT NULL,
    Sueldo FLOAT
)TABLESPACE TiendaCrecimientoLento STORAGE (INITIAL 5K NEXT 20K);

CREATE TABLE CLIENTE (
    dniCl VARCHAR2(9) PRIMARY KEY,
    Nombre VARCHAR2(12) NOT NULL,
    prApellido VARCHAR2(20) NOT NULL,
    sgApellido VARCHAR2(20),
    Direccion VARCHAR2(15) NOT NULL,
    Telefono NUMBER(9) NOT NULL,
    Correo VARCHAR2(20) NOT NULL,
    Ciudad VARCHAR2(10) NOT NULL
)TABLESPACE TiendaCrecimientoConstante STORAGE (INITIAL 11K NEXT 30K);

CREATE TABLE VENTA(
    codVenta NUMBER(5) PRIMARY KEY,
    fechaHora DATE,
    dniEmp VARCHAR2(9) NOT NULL,
    dniCli VARCHAR2(9) NOT NULL,
    FOREIGN KEY (dniEmp) REFERENCES EMPLEADO(dniEmpl),
    FOREIGN KEY (dniCli) REFERENCES CLIENTE(dniCl)
)TABLESPACE TiendaCrecimientoConstante STORAGE(INITIAL 12K NEXT 70K);

CREATE TABLE COMPRA(
    codCompra NUMBER(5) PRIMARY KEY,
    fechaHora DATE,
    dniEmp VARCHAR2(9) NOT NULL,
    codProv NUMBER(2) NOT NULL,
    FOREIGN KEY (dniEmp) REFERENCES EMPLEADO(dniEmpl),
    FOREIGN KEY (codProv) REFERENCES PROVEEDOR(codProv)
)TABLESPACE TiendaCrecimientoConstante STORAGE(INITIAL 20K NEXT 65K);

CREATE TABLE LINEAVENTA(
    codVenta NUMBER(5),
    codProducto NUMBER(4),
    cant NUMBER(3),
    PRIMARY KEY(codVenta,codProducto),
    FOREIGN KEY (codVenta) REFERENCES VENTA(codVenta),
    FOREIGN KEY (codProducto) REFERENCES PRODUCTO(codProducto)
)TABLESPACE TiendaCrecimientoConstante STORAGE(INITIAL 22K NEXT 70K);

CREATE TABLE LINEACOMPRA(
    codCompra NUMBER(5),
    codProducto NUMBER(4),
    cant NUMBER(3),
    PRIMARY KEY(codCompra,codProducto),
    FOREIGN KEY (codCompra) REFERENCES COMPRA(codCompra),
    FOREIGN KEY (codProducto) REFERENCES PRODUCTO(codProducto)
)TABLESPACE TiendaCrecimientoConstante STORAGE(INITIAL 22K NEXT 70K);
